Descripcion
MCU   = atmega168pa
F_CPU = 1000000UL
BAUD  = 9600UL

PROGRAMMER = usbtiny

CC = avr-gcc
OBJCOPY = avr-objcopy
AVRDUDE = avrdude

CODE_FILES = led_blink.c

TARGET = led_blink


$(TARGET).hex: $(TARGET).elf
	$(OBJCOPY) -O ihex -R .eeprom $(TARGET).elf $(TARGET).hex

$(TARGET).elf: $(CODE_FILES)
	$(CC) -mmcu=$(MCU) -Os -o $(TARGET).elf $(CODE_FILES)

flash: $(TARGET).hex
	$(AVRDUDE) -c $(PROGRAMMER) -p $(MCU) -U flash:w:$(TARGET).hex:i
